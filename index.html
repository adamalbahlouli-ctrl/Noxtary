-- 001_create_tables.sql -- Create main apps/content table CREATE TABLE IF NOT EXISTS apps ( id BIGSERIAL PRIMARY KEY, app_id TEXT UNIQUE NOT NULL, type TEXT NOT NULL CHECK (type IN ('apps','books','articles','mods','audio','services')), title TEXT NOT NULL, description TEXT, "longDesc" TEXT, image TEXT, "downloadUrl" TEXT, category TEXT, version TEXT, size TEXT, author TEXT, pages INTEGER, read_time TEXT, publish_date TEXT, duration TEXT, audio_type TEXT, game TEXT, price TEXT, screenshots JSONB DEFAULT '[]', created_at TIMESTAMPTZ DEFAULT NOW() );
